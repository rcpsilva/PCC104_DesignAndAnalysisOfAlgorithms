    \begin{itemize}

    \end{itemize}
    \end{block}

\begin{frame}{Problema Computacional}

\begin{block}{Definição}
Um problema computacional é a especificação de uma tarefa que define,
para cada valor de \og{entrada}, quais valores de \og{saída} são considerados corretos.
\end{block}

\vspace{0.4cm}

\begin{exampleblock}{Exemplo: Ordenação}

\textbf{Entrada:} uma sequência de números  
\[
(a_1, a_2, \dots, a_n)
\]

\textbf{Saída:} uma permutação da sequência tal que

\[
a_1 \le a_2 \le \dots \le a_n
\]

\end{exampleblock}

\end{frame}
%---------------------------------------
\begin{frame}{Instância de um Problema}

\begin{block}{Definição}
Uma \og{instância de um problema} é um conjunto específico de valores
de entrada para um problema computacional.
\end{block}

\vspace{0.4cm}

\begin{exampleblock}{Exemplo: Problema de Ordenação}

\textbf{Problema:} ordenar uma sequência de números.

\textbf{Instância:}

\[
(8,\,3,\,5,\,2,\,9)
\]

\end{exampleblock}

\end{frame}

\begin{frame}{Ordenação por Inserção (Insertion Sort)}

\begin{block}{Problema de Ordenação}
\textbf{Entrada:}  
Uma sequência de $n$ números  
\[
\langle a_1, a_2, \dots, a_n \rangle
\]

\vspace{0.3cm}

\textbf{Saída:}  
Uma permutação (reordenação) da sequência de entrada  
\[
\langle a_1', a_2', \dots, a_n' \rangle
\]
tal que  
\[
a_1' \leq a_2' \leq \dots \leq a_n'
\]

\end{block}

\vspace{0.3cm}

\begin{alertblock}{Objetivo}
Reordenar os elementos da entrada em ordem não decrescente.
\end{alertblock}

\end{frame}
% ---------------------------------------------
\begin{frame}{Ordenação por Inserção}

\begin{block}{\textsc{Insertion-Sort}$(A)$}
\begin{algorithmic}[1]
\For{$j \gets 2$ \textbf{to} $A.\text{length}$}
    \State $key \gets A[j]$
    \State \Comment{Insere $A[j]$ na sequência ordenada $A[1 \dots j-1]$}
    \State $i \gets j - 1$
    \While{$i > 0$ \textbf{and} $A[i] > key$}
        \State $A[i + 1] \gets A[i]$
        \State $i \gets i - 1$
    \EndWhile
    \State $A[i + 1] \gets key$
\EndFor
\end{algorithmic}
\end{block}

\og{Exemplo} : $[5,2,4,6,1,3]$

\end{frame}
% ---------------------------------------------
\begin{frame}{Corretude da ordenação por inserção}

\begin{block}{Invariante de Laço}
Uma \textbf{invariante de laço} é uma propriedade que deve ser verdadeira:
\begin{itemize}
    \item \textbf{Inicialização:} antes da primeira iteração do laço;
    \item \textbf{Manutenção:} antes de cada iteração, permanece verdadeira após a execução do corpo do laço;
    \item \textbf{Término:} ao final do laço, fornece uma propriedade útil para demonstrar a correção do algoritmo.
\end{itemize}
\end{block}

\vspace{0.3cm}

\begin{alertblock}{Uso}
Invariantes de laço são utilizadas para demonstrar a \textbf{correção} de algoritmos iterativos.
\end{alertblock}

\end{frame}
%----------------------------------------------
\begin{frame}{Correção do Insertion Sort}

\begin{block}{Invariante de Laço}
No início de cada iteração do laço \textbf{for} (linha 1), o subarray
\[
A[1 \dots j-1]
\]
consiste nos elementos originalmente em $A[1 \dots j-1]$, porém em ordem não decrescente.
\end{block}

\begin{block}{Prova}
\textbf{Inicialização:}  
Para $j = 2$, o subarray $A[1]$ contém apenas um elemento, logo está ordenado.

\vspace{0.2cm}

\textbf{Manutenção:}  
A cada iteração, o elemento $A[j]$ é inserido na posição correta em $A[1 \dots j-1]$, produzindo $A[1 \dots j]$ ordenado.

\vspace{0.2cm}

\textbf{Término:}  
Quando o laço termina ($j = A.\text{length} + 1$), temos
\[
A[1 \dots A.\text{length}]
\]
ordenado.
\end{block}

\end{frame}

\begin{frame}{O que é Análise de Algoritmos?}

\begin{block}{Definição}
A \textbf{análise de algoritmos} consiste em prever os recursos computacionais necessários para executar um algoritmo, tipicamente em função do tamanho da entrada $n$.
\end{block}

\vspace{0.3cm}

\begin{block}{Por que analisar algoritmos?}
\begin{itemize}
    \item Comparar diferentes algoritmos para o mesmo problema;
    \item Prever desempenho para entradas grandes;
    \item Compreender como o tempo de execução cresce com $n$;
    \item Guiar o projeto de algoritmos mais eficientes.
\end{itemize}
\end{block}

\vspace{0.3cm}

\begin{alertblock}{Desafio}
Para analisar um algoritmo, precisamos de um modelo que defina:
\begin{itemize}
    \item O que é uma operação;
    \item Quanto custa cada operação.
\end{itemize}
\end{alertblock}

\vspace{0.2cm}

\centering
$\Rightarrow$ \textbf{Modelo de Computação (RAM)}

\end{frame}
%--------------------------------------------
\begin{frame}{Modelo de Computação (RAM -\textit{Random-access Machine})}


Na análise de algoritmos, assumimos o modelo RAM, no qual:
\begin{itemize}
    \item As instruções são executadas sequencialmente;
    \item Cada operação básica (atribuição, soma, comparação, acesso a memória) custa tempo constante;
    \item O acesso à memória é uniforme (tempo constante para qualquer posição).
\end{itemize}


\begin{block}{Cuidados}
\begin{itemize}
    \item O custo constante pode variar entre operações (ex.: divisão vs. soma);
    \item Ignoramos efeitos de hardware (cache, paralelismo, I/O);
    \item Assumimos que os dados cabem na memória principal.
\end{itemize}
\end{block}

\begin{alertblock}{Limitações}
\begin{itemize}
    \item Não modela hierarquia de memória;
    \item Não captura custo de operações com números muito grandes;
    \item Pode não refletir desempenho real em sistemas modernos.
\end{itemize}
\end{alertblock}

\end{frame}

\begin{itemize}

\end{itemize}
\end{alertblock}

\begin{frame}{Insertion Sort: Custo e Frequência}

\small

\begin{columns}[T]

%----------------------------------
\begin{column}{0.55\textwidth}

\textbf{Ordena-Por-Inserção$(A,n)$}

\begin{enumerate}
    \item \textbf{for} $i = 2$ \textbf{to} $n$
    \item \quad chave $= A[i]$
    \item \quad $j = i - 1$
    \item \quad \textbf{while} $j > 0$ \textbf{and} $A[j] >$ chave
    \item \qquad $A[j+1] = A[j]$
    \item \qquad $j = j - 1$
    \item \quad $A[j+1] =$ chave
\end{enumerate}

\end{column}

%----------------------------------
\begin{column}{0.45\textwidth}

\textbf{Custo: n$^o$ Vezes}

\vspace{0.2cm}

\begin{itemize}
    \item $c_1$ : $n - 1$ 
    \item $c_2$ : $n - 1$ 
    \item $c_3$ : $n - 1$ 
    \item $c_4$ : $n - 1$ 
    \item $c_5$ : $\sum_{i=2}^{n} (t_i - 1)$ 
    \item $c_6$ : $\sum_{i=2}^{n} (t_i - 1)$ 
    \item $c_7$ : $n - 1$ 
\end{itemize}

\end{column}

%----------------------------------
\end{columns}

\vspace{0.3cm}

\begin{alertblock}{Observação}
$t_i$ é o número de vezes que o teste do \texttt{while} é executado para cada $i$.
\end{alertblock}

\end{frame}

\begin{frame}{Casos de Análise de Algoritmos}

\begin{block}{Melhor Caso}
\begin{itemize}
    \item Menor tempo de execução para entradas de tamanho $n$
    \item Ocorre quando a entrada favorece o algoritmo
    \item Formalmente:
    \[
    T_{\text{best}}(n) = \min_{\text{entradas de tamanho } n} T(n)
    \]
\end{itemize}
\end{block}

\end{frame}
%-------------------------------------------------
\begin{frame}{Casos de Análise de Algoritmos}


\begin{block}{Pior Caso}
\begin{itemize}
    \item Maior tempo de execução para entradas de tamanho $n$
    \item Garante limite superior de desempenho
    \item Formalmente:
    \[
    T_{\text{worst}}(n) = \max_{\text{entradas de tamanho } n} T(n)
    \]
\end{itemize}
\end{block}

\vspace{0.2cm}


\end{frame}
%------------------------------------------------
\begin{frame}{Casos de Análise de Algoritmos}

\begin{block}{Caso Médio}
\begin{itemize}
    \item Tempo esperado sobre todas as entradas de tamanho $n$
    \item Depende de uma distribuição de probabilidade
    \item Formalmente:
    \[
    T_{\text{avg}}(n) = \mathbb{E}[T(n)]
    \]
\end{itemize}
\end{block}

\end{frame}
%------------------------------------------------
\begin{frame}{Por que analisar o pior caso?}

\begin{block}{Garantia de Desempenho}
\begin{itemize}
    \item Fornece um limite superior para o tempo de execução
    \item Garante que o algoritmo nunca será pior do que isso
    \item \og{O caso médio é quase tão ruim quanto o pior caso}
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Independência da Entrada}
\begin{itemize}
    \item Não depende de hipóteses sobre a distribuição dos dados
    \item Evita suposições difíceis de justificar na prática
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Robustez}
\begin{itemize}
    \item Essencial em sistemas críticos (tempo real, segurança, etc.)
    \item Protege contra entradas adversariais
\end{itemize}
\end{block}


\end{frame}

\begin{frame}{Por que não contar exatamente o número de operações?}

\begin{block}{Constantes e Detalhes Irrelevantes}
\begin{itemize}
    \item Diferenças como:
    \[
    T(n) = 3n^2 + 10n + 5 \quad \text{vs} \quad T(n) = n^2
    \]
    \item Para $n$ grande, o termo dominante é o mesmo
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Baixa Escalabilidade da Análise Exata}
\begin{itemize}
    \item Expressões exatas são complexas e pouco informativas
    \item Difíceis de comparar entre algoritmos
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Ideia Central}
\begin{itemize}
    \item Queremos capturar o comportamento para $n$ grande
    \item Por isso usamos \textbf{taxa de crescimento} e \textbf{notação assintótica}
\end{itemize}
\end{alertblock}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $O$ (Big-O)}

\begin{block}{Definição Intuitiva}
\begin{itemize}
    \item A notação $O$ caracteriza um \textbf{limite superior} assintótico
    \item Indica que uma função não cresce mais rápido que uma certa taxa
    \item Baseia-se no \textbf{termo dominante} (maior ordem)
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Exemplo}
\[
T(n) = 7n^3 + 100n^2 - 20n + 6
\]

\begin{itemize}
    \item Termo dominante: $n^3$
    \item Logo: $T(n) = O(n^3)$
\end{itemize}
\end{block}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $O$ (Big-O)}

\begin{block}{Observação Importante}
\begin{itemize}
    \item Também podemos escrever:
    \[
    T(n) = O(n^4),\; O(n^5), \dots
    \]
    \item Pois a função cresce mais lentamente que essas taxas
    \item Em geral: $T(n) = O(n^c)$ para qualquer $c \geq 3$
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Mensagem-chave}
A notação $O$ fornece um \textbf{limite superior}, não necessariamente o mais justo.
\end{alertblock}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $\Omega$ (Big-Omega)}

\begin{block}{Definição Intuitiva}
\begin{itemize}
    \item A notação $\Omega$ caracteriza um \textbf{limite inferior} assintótico
    \item Indica que uma função cresce \textbf{pelo menos tão rápido} quanto uma certa taxa
    \item Foca no comportamento para $n$ \textbf{grande}
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Exemplo}
\[
T(n) = 7n^3 + 100n^2 - 20n + 6
\]

\begin{itemize}
    \item Termo dominante: $n^3$
    \item Logo: $T(n) = \Omega(n^3)$
\end{itemize}
\end{block}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $\Omega$ (Big-Omega)}

\begin{block}{Observação Importante}
\begin{itemize}
    \item Também podemos escrever:
    \[
    T(n) = \Omega(n^2),\; \Omega(n), \dots
    \]
    \item Pois a função cresce pelo menos tão rápido quanto essas taxas
    \item Em geral: $T(n) = \Omega(n^c)$ para qualquer $c \leq 3$
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Interpretação}
\begin{itemize}
    \item $\Omega(\cdot)$ fornece um \textbf{limite inferior}
    \item Descreve o \textbf{melhor crescimento possível} do algoritmo
\end{itemize}
\end{alertblock}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $\Theta$ (Big-Theta)}

\begin{block}{Definição Intuitiva}
\begin{itemize}
    \item A notação $\Theta$ caracteriza um \textbf{limite assintótico justo}
    \item Indica que a função cresce \textbf{exatamente} a uma certa taxa
    \item Considera limites superior e inferior simultaneamente
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Interpretação}
\begin{itemize}
    \item A função está limitada por constantes:
    \[
    c_1 f(n) \leq T(n) \leq c_2 f(n)
    \]
    \item Para $n$ suficientemente grande
    \item Os fatores $c_1$ e $c_2$ não precisam ser iguais
\end{itemize}
\end{block}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $\Theta$ (Big-Theta)}

\begin{block}{Caracterização}
\begin{itemize}
    \item Se:
    \[
    T(n) = O(f(n)) \quad \text{e} \quad T(n) = \Omega(f(n))
    \]
    \item Então:
    \[
    T(n) = \Theta(f(n))
    \]
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Exemplo}
\[
T(n) = 7n^3 + 100n^2 - 20n + 6
\]

\begin{itemize}
    \item $T(n) = O(n^3)$ e $T(n) = \Omega(n^3)$
    \item Logo: $T(n) = \Theta(n^3)$
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Mensagem-chave}
$\Theta(\cdot)$ descreve a \textbf{taxa de crescimento exata} do algoritmo.
\end{alertblock}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $O$ — Definição Formal}

\begin{block}{Ideia}
\begin{itemize}
    \item A notação $O$ descreve um \textbf{limite assintótico superior}
    \item Fornece um limite dentro de um \textbf{fator constante}
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Definição Formal}
Para uma função $g(n)$, definimos:
\small
\[
O(g(n)) = \left \{ f(n) : \exists\, c > 0,\; \exists\, n_0 > 0 \text{ tais que } 0 \leq f(n) \leq c\,g(n),\; \forall n \geq n_0 \right \} 
\]
\end{block}
\normalsize
\vspace{0.2cm}

\begin{alertblock}{Interpretação}
\begin{itemize}
    \item A função $f(n)$ não cresce mais rápido que $g(n)$
    \item A desigualdade vale a partir de algum ponto $n_0$
\end{itemize}
\end{alertblock}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Exemplo: Prova usando a definição formal de $O$}
\small
\begin{block}{Objetivo}
Mostrar que:
\[
f(n) = 3n^2 + 5n + 2 \;\;\text{é}\;\; O(n^2)
\]
\end{block}

\vspace{0.2cm}

\begin{block}{Passo 1: Majorar a função}
Para $n \geq 1$:
\[
3n^2 + 5n + 2 \leq 3n^2 + 5n^2 + 2n^2 = 10n^2
\]
\end{block}

\vspace{0.2cm}

\begin{block}{Passo 2: Escolha das constantes}
\begin{itemize}
    \item $c = 10$
    \item $n_0 = 1$
\end{itemize}

Logo:
\[
0 \leq f(n) \leq 10n^2 \quad \forall n \geq 1
\]
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Conclusão}
Pela definição formal, $f(n) = O(n^2)$.
\end{alertblock}

\end{frame}

\begin{frame}{Notação $\Omega$ — Definição Formal}

\begin{block}{Ideia}
\begin{itemize}
    \item A notação $\Omega$ descreve um \textbf{limite assintótico inferior}
    \item Garante crescimento mínimo dentro de um \textbf{fator constante}
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Definição Formal}
Para uma função $g(n)$, definimos:

\small
\[
\Omega(g(n)) = \left\{ f(n) : \exists\, c > 0,\; \exists\, n_0 > 0 \text{ tais que } 
0 \leq c\,g(n) \leq f(n),\; \forall n \geq n_0 \right\}
\]
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Interpretação}
\begin{itemize}
    \item A função $f(n)$ cresce pelo menos tão rápido quanto $g(n)$
    \item A desigualdade vale a partir de algum ponto $n_0$
\end{itemize}
\end{alertblock}

\end{frame}
%---------------------------------------------------------
\begin{frame}{Notação $\Theta$ — Definição Formal}

\begin{block}{Ideia}
\begin{itemize}
    \item A notação $\Theta$ descreve um \textbf{limite assintótico justo}
    \item Combina limite superior e inferior
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{block}{Definição Formal}
Para uma função $g(n)$, definimos:

\[
\Theta(g(n)) = \{ f(n) : \exists\, c_1, c_2 > 0,\; \exists\, n_0 > 0 \text{ tais que }
\]
\[
0 \leq c_1 g(n) \leq f(n) \leq c_2 g(n),\; \forall n \geq n_0 \}
\]
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Interpretação}
\begin{itemize}
    \item $f(n)$ está limitada superior e inferiormente por $g(n)$
    \item Descreve a \textbf{taxa de crescimento exata}
\end{itemize}
\end{alertblock}

\end{frame}

\begin{frame}{Comparação de Funções Assintóticas}

    \item Se $f(n) = O(g(n))$ e $g(n) = O(h(n))$, então $f(n) = O(h(n))$
    \item Se $f(n) = \Omega(g(n))$ e $g(n) = \Omega(h(n))$, então $f(n) = \Omega(h(n))$
    \item Se $f(n) = o(g(n))$ e $g(n) = o(h(n))$, então $f(n) = o(h(n))$
    \item Se $f(n) = \omega(g(n))$ e $g(n) = \omega(h(n))$, então $f(n) = \omega(h(n))$
\end{itemize}
\end{block}

\vspace{0.2cm}

\begin{alertblock}{Ideia}
As relações assintóticas preservam a ordem de comparação entre funções.
\end{alertblock}

\begin{frame}{Comparação de Funções Assintóticas}
\small
\begin{block}{Reflexividade}
\[
f(n) = \Theta(f(n)), \qquad
f(n) = O(f(n)), \qquad
f(n) = \Omega(f(n))
\]
\end{block}


\begin{block}{Simetria}
\[
f(n) = \Theta(g(n))
\iff
g(n) = \Theta(f(n))
\]
\end{block}



\begin{block}{Simetria de Transposição}
\[
f(n) = O(g(n))
\iff
g(n) = \Omega(f(n))
\]

\[
f(n) = o(g(n))
\iff
g(n) = \omega(f(n))
\]
\end{block}


\begin{alertblock}{Mensagem-chave}
$\Theta$ é simétrica; $O$ e $\Omega$ são relações transpostas uma da outra.
\end{alertblock}
